이분탐색

주어지는 k개의 값의 최대값이 2^31 - 1 즉, 2147483647이다.
따라서 가능한 랜선의 최대길이는 1\~2147483647이다.

이분탐색을 이용하여 반씩 쪼개가면서 해당 값으로 몇개의
랜선이 만들어지는지 탐색하는 과정을 약 32번 하면 원하는
결과가 나오게 된다.
(32번을 반복한 이유는 2147483647를 32번 반으로 쪼개면
1보다 작아져서 충분한 탐색이 가능하기 때문이다.
\= 원하는 정수 도출 가능).

C

```c
#include <stdio.h>

int main(){
    int k, n; scanf("%d %d", &k, &n);
    int v[10000];
    for(int i = 0; i < k; i++) scanf("%d", &v[i]);

    long long res = 2147483647;
    long long t1 = 0, t2 = res;
    int t = 32; 
    while(t--){
        long long sum = 0;
        if(res == 0) res = 1;
        for(int i = 0; i < k; i++) sum += v[i] / res;
        
        if(sum < n){
            t2 = res; 
            res = (res + t1) / 2;
        }
        else{
            t1 = res; 
            res = (res + t2) / 2;
        }
    }
    
    // 최종 결과 검증 (결과값이 n개 이상을 만족하는지 확인)
    long long final_sum = 0;
    for(int i = 0; i < k; i++) final_sum += v[i] / res;
    if(final_sum < n) res--;
    
    printf("%lld", res);
    return 0;
}
```

Python

```python
import sys

k, n = map(int, sys.stdin.readline().split())
v = [int(sys.stdin.readline()) for _ in range(k)]

res = 2147483647
t1 = 0
t2 = res

for _ in range(32):
    if res == 0: res = 1
    total = 0
    for length in v:
        total += length // res
        
    if total < n:
        t2 = res
        res = (res + t1) // 2
    else:
        t1 = res
        res = (res + t2) // 2

# 최종 결과 검증
final_total = 0
for length in v:
    final_total += length // res
if final_total < n:
    res -= 1

print(res)
```
