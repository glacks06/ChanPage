가장 큰 단위의 z 부터 가장 작은 단위(2\*2)의 z까지 재귀함수를 통해 쪼개는 문제.
z의 시작점들을 기준으로 작은 단위로 쪼개간다. 이때 (r, c)가 포함이 되는 범위인지 확인하고
포함이 안된다면 연산 횟수 감소를 위해 해당 범위의 칸 개수만큼을 결과에 더하고 그냥 넘어간다.
마지막에 (r, c)에 도달하면 그동안 지나온 칸들을 더한 값을 출력한다.

C

```c
#include <stdio.h>

int visit(int px, int py, int* index, int n, int r, int c){
    if(!(px <= r && r <= (px+(1<<n))-1) || !(py <= c && c <= (py+(1<<n))-1)){ // 범위에 r, c 안들어가면 걸러짐
        *index += (1<<n) * (1<<n); return 0;
    }
    
    if(n == 0) printf("%d", *index);
    else{
        visit(px, py, &(*index), n-1, r, c);
        visit(px, py + (1<<n)/2, &(*index), n-1, r, c);
        visit(px + (1<<n)/2, py, &(*index), n-1, r, c);
        visit(px + (1<<n)/2, py + (1<<n)/2, &(*index), n-1, r, c);
    }
    return 0;
}

int main() {
    int n, r, c;
    scanf("%d %d %d", &n, &r, &c);
    int index = 0;
    visit(0, 0, &index, n, r, c);
}
```

def visit(px, py, n):
    global r, c, index
    if not(px <= r and r <= (px+(2**n))-1) or not(py <= c and c <= (py+(2**n))-1):
        index += (2**n)**2
        return 0
    if n == 0: print(index)
    else:
        visit(px, py, n-1)
        visit(px, py + (2**n)//2, n-1)
        visit(px + (2**n)//2, py, n-1)
        visit(px + (2**n)//2, py + (2**n)//2, n-1)
    return 0


n, r, c = map(int, input().split(' '))
index = 0
visit(0, 0, n)
