[https://www.acmicpc.net/problem/1780](https://www.acmicpc.net/problem/1780)

만일 전체 배열의 값이 동일하지 않으면 종이를 9등분을 한 후 각각의 2차원 배열들이 동일한 값을 가지는지 검사를 하는 방식으로 문제를 해결하였다.
이 과정에서 재귀함수를 사용해야겠다는 판단을 하였다.

우선 전체 배열을 v에 저장하고, f함수에서 해당 배열의 값들이 전부 동일한지 same함수를 이용해 검사한다.
만일 동일하다면 동일한 값에 해당하는 변수에 1을 더한다.(numM1 = -1, num0 = 0, num1 = 1)
동일하지 않다면 배열을 9등분하여 다시 검사한다.(재귀)
검사하는 배열의 크기가 1x1이 될때까지 반복한 후 numM1, num0, num1의 값을 출력한다.

C

```c
#include <stdio.h>

int v[2187][2187];
int num1, num0, numM1;

int same(int x, int y, int r){
    int a = v[x][y];
    for(int i = x; i < x + r; i++){
        for(int j = y; j < y + r; j++){
            if(a != v[i][j]) return 0; // 범위 내 숫자 동일x
        }
    }
    if(a == 1) num1++;
    else if(a == 0) num0++;
    else if(a == -1) numM1++;
    return 1; // 범위 내 숫자 동일o
}

void f(int r, int start_x, int start_y){
    // 현재 범위가 모두 같은 숫자가 아니면 9등분하여 재귀 호출
    if(!same(start_x, start_y, r)){
        int next_r = r / 3;
        for(int i = 0; i < 3; i++){
            for(int j = 0; j < 3; j++){
                f(next_r, start_x + i * next_r, start_y + j * next_r);
            }
        }
    }
}

int main(){
    int n;
    if(scanf("%d", &n) != 1) return 0;
    for(int i = 0; i < n; i++){
        for(int j = 0; j < n; j++){
            scanf("%d", &v[j][i]);
        }
    }
    
    f(n, 0, 0);
    printf("%d\n%d\n%d", numM1, num0, num1);
    return 0;
}
```

Python

```python
import sys

# 재귀 깊이 제한 해제
sys.setrecursionlimit(10**6)

def same(x, y, r):
    global num1, num0, numM1
    a = v[x][y]
    for i in range(x, x + r):
        for j in range(y, y + r):
            if a != v[i][j]:
                return False
    if a == 1: num1 += 1
    elif a == 0: num0 += 1
    else: numM1 += 1
    return True

def f(r, start_x, start_y):
    if not same(start_x, start_y, r):
        next_r = r // 3
        for i in range(3):
            for j in range(3):
                f(next_r, start_x + i * next_r, start_y + j * next_r)

n = int(sys.stdin.readline())
num1, num0, numM1 = 0, 0, 0
v = []
for _ in range(n):
    v.append(list(map(int, sys.stdin.readline().split())))

# 입력된 v가 [row][col] 형식이므로 좌표 처리에 유의 (여기서는 v[x][y]로 접근)
# 기존 코드의 입력을 Python 리스트 형식에 맞춰 수정하여 탐색
def same_refactored(x, y, r):
    global num1, num0, numM1
    first = v[x][y]
    for i in range(x, x + r):
        for j in range(y, y + r):
            if v[i][j] != first:
                return False
    if first == 1: num1 += 1
    elif first == 0: num0 += 1
    else: numM1 += 1
    return True

def f_refactored(r, x, y):
    if not same_refactored(x, y, r):
        nr = r // 3
        for i in range(3):
            for j in range(3):
                f_refactored(nr, x + i * nr, y + j * nr)

f_refactored(n, 0, 0)
print(f"{numM1}\n{num0}\n{num1}")
```
