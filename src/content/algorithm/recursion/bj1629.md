[https://www.acmicpc.net/problem/1629](https://www.acmicpc.net/problem/1629)

a^b = a*a*a...

a들을 반씩 나누어서 계산 후 c로 나눠가며 나머지를 구하는 함수(b==1 -\> a%c)
\-\> 재귀
결과 출력

C

```c
#include <stdio.h>

long long multiple(long long a, long long b, long long c){
    if(b == 1) return a % c;
    
    long long t = multiple(a, b / 2, c);
    t = (t * t) % c;
    
    if(b % 2 == 0) return t;
    else return (t * (a % c)) % c;
}

int main() {
    long long a, b, c;
    scanf("%lld %lld %lld", &a, &b, &c);
    printf("%lld", multiple(a, b, c));
    return 0;
}
```

<br>

Python

```python
def multiple(a, b, c):
    if b == 1:
        return a % c
    
    t = multiple(a, b // 2, c)
    t = (t * t) % c
    
    if b % 2 == 0:
        return t
    else:
        return (t * (a % c)) % c

a, b, c = map(int, input().split())
print(multiple(a, b, c))
```
