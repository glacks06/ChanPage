최대 공약수가 주어졌으므로 최대 공약수 값부터 시작하여 최대 공약수의 배수들을 최소공배수에 나누어 생긴 두 숫자의 최대 공약수와 최소공배수를 계산하여 주어진 최대 공약수와 최소 공배수와 일치하는지 확인하였다. 

만약 일치한다면 해당 숫자의 합이 조건을 만족하는 다른 숫자의 합 중 최솟값보다 작다면 최솟값을 갱신한다. 

확인이 다 끝난 후 결과로 나온 값(minJ, minI)를 출력한다.

C

```c
#include <stdio.h>

int divisor(long a, long b){
    long r;
    while(b){r = a % b; a = b; b = r;}
    return a;
}

int multiple(long a, long b){
    long add = (a > b) ? a : b;
    long c = (a < b) ? a : b;
    long res;
    for(long i = add; i <= a * b; i += add){
        res = i;
        if(i % c == 0) break;
    }
    return res;
}

int main(){
    long a, b; scanf("%ld %ld", &a, &b);
    long minI = 1000000000, minJ = 1000000000;
    long i = 0, j;
    while(i <= b){
        i += a; j = (a*b) / i;
        if(j % a != 0) continue;
        long m = multiple(i, j);
        long d = divisor(i, j);
        if(d == a && m == b){
            if(i + j < minI + minJ){minI = i; minJ = j;}
            else break;
        }
    }
    if(minI < minJ) printf("%ld %ld", minI, minJ);
    else printf("%ld %ld", minJ, minI);
}
```

def multiple(a, b):
    add = a if a > b else b
    c = a if a < b else b
    res = 0
    for i in range(add, (a*b)+1, add):
        res = i
        if res % c == 0: break
    return res

def divisor(a, b):
    r = 0
    while b:
        r = a % b; a = b; b = r
    return a

a, b = map(int, input().split())
i = 0; minI = 1000000000; minJ = 1000000000

while i <= b:
    i += a; j = (a*b) // i
    if j % a != 0: continue

    m = multiple(i, j)
    d = divisor(i, j)

    if d == a and m == b:
        if i + j < minI + minJ:
            minI = i; minJ = j
        else: break

print(minI, minJ)
