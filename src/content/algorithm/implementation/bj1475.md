0\~8까지에 해당하는 각각의 공간이 있는 배열을 생성한다. (각 배열의 인덱스는 필요한 해당 숫자의 플라스틱 숫자의 개수이다.)
주어진 값의 각 자릿수를 분리한 후
각 자릿수에 해당하는 숫자를 하나씩 확인.
앞에 만든 배열에서 확인한 숫자에 해당하는 인덱스에 +1(만약 숫자가 9면  일단은 6에 +1)
6과 9는 뒤집어서 쓸 수 있으므로 6에 해당하는 인덱스의 값을 2로 나눈 후 배열의 원소 중 가장 큰 값 출력.

C

```c
#include <stdio.h>

int main(){
    int n; scanf("%d", &n);
    int num[9] = {0,};
    while(n){
        int idx = (n%10 == 9) ? 6 : n % 10;
        num[idx]++;
        n /= 10;
    }
    num[6] = 1 + (num[6]-1) / 2;
    int max = 0;
    for(int i = 0; i < 9; i++) if(num[i] > max) max = num[i];
    printf("%d", max);
}
```

n = int(input())
num = [0 for _ in range(9)]
while n:
    idx = 6 if n % 10 == 9 else n % 10
    num[idx] += 1
    n //= 10

for i in range(9): 
    if num[i] > max: max = num[i]
