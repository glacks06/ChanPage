[https://www.acmicpc.net/problem/2447](https://www.acmicpc.net/problem/2447)

별이 공백 주변을 감싸고, 그 모양이 공백을 주변을 감싸는 패턴이다. 이를 재귀함수를 통해 해결하였다. 공백으로 채워진 N*N크기의 이차원 배열에서 공간을 9개로 나누고, 이를 다시 9개로 나누는 방식으로 3*3크기의 2차원 배열이 될대까지 나눈 후 정 가운데를 제외한 나머지를 별로 채웠다.

C

```c
#include <stdio.h>

char arr[2187][2187];

void prints(int n){
    for(int i = 0; i < n; i++){
        for(int j = 0; j < n; j++) printf("%c", arr[i][j]);
        printf("\n");
    }
}

void printStar(int pn, int pm){
    for(int i = 0; i < 3; i++)
        arr[pn][pm+i] = '*';
    for(int i = 0; i < 3; i++)
        arr[pn+i][pm] = '*';
    for(int i = 0; i < 3; i++)
        arr[pn+2][pm+i] = '*';
    for(int i = 0; i < 3; i++)
        arr[pn+i][pm+2] = '*';
}

int f(int n, int pn, int pm){
    if(n == 3){
        printStar(pn, pm); return 0;
    }
    else{
        f(n/3, pn, pm); // 왼 위
        f(n/3, pn, pm+(n/3)); // 중앙 위
        f(n/3, pn, pm+(2*(n/3))); // 오른 위
        f(n/3, pn+(n/3), pm); //왼 중앙
        f(n/3, pn+(2*(n/3)), pm); // 왼 아래
        f(n/3, pn+(n/3), pm+(2*(n/3))); // 오른 중앙
        f(n/3, pn+(2*(n/3)), pm+(n/3)); // 중앙 아래
        f(n/3, pn+(2*(n/3)), pm+(2*(n/3))); // 오른 아래
    }
    return 0;
}

int main() {
    int n;
    scanf("%d", &n);
    for(int i = 0; i < n; i++)
        for(int j = 0; j < n; j++) arr[i][j] = ' ';
    f(n, 0, 0);
    prints(n);
}
```

```python
arr = [[' ' for j in range(2187)] for i in range(2187)]

def prints(n):
    for i in range(n):
        for j in range(n): print(arr[i][j], end='')
        print()

def printStar(pn, pm):
    for i in range(3):
        arr[pn][pm+i] = '*'
    for i in range(3):
        arr[pn+i][pm] = '*'
    for i in range(3):
        arr[pn+2][pm+i] = '*'
    for i in range(3):
        arr[pn+i][pm+2] = '*'

def f(n, pn, pm):
    if n == 3:
        printStar(pn, pm)
    else:
        f(n//3, pn, pm)
        f(n//3, pn, pm+(n//3))
        f(n//3, pn, pm+(2*(n//3)))
        f(n//3, pn+(n//3), pm)
        f(n//3, pn+(2*(n//3)), pm)
        f(n//3, pn+(n//3), pm+(2*(n//3)))
        f(n//3, pn+(2*(n//3)), pm+(n//3))
        f(n//3, pn+(2*(n//3)), pm+(2*(n//3)))

n = int(input())
f(n, 0, 0)
prints(n)
```
