[https://www.acmicpc.net/problem/1094](https://www.acmicpc.net/problem/1094)

64cm 막대기를 절반씩 잘랐을 때 나올 수 있는 길이의 종류는 2cm, 4cm, 8cm, 16cm, 32cm 즉, 2의 제곱수이다.
주어진 x값을 이진수로 변환하면 어떤 2의 제곱수(막대기)의 합인지 알 수 있다.
게다가 이진수로 변환하였을 때 1의 개수가 자연스럽게 최소한으로 막대기를 잘랐을 때의 막대 개수가 되므로 값이 1인 자릿수의 갯수를 구하여 출력한다.

C

```c
#include <stdio.h>

int binary(int v){ // 이진수로 변환
    int res = 0, i = 1;
    while(v){
        res += (v % 2) * i;
        v = v / 2; i *= 10;
    }
    return res;
}

int main(){
    int x; scanf("%d", &x);
    int v = binary((x/2) * 2);
    int cnt = 0;
    while(v){
        if(v % 10 == 1) cnt++;
        v /= 10;
    }
    printf("%d", cnt + (x % 2));
}
```

```python
def binary(v): # 이진수로 변환
    res = 0; i = 1
    while v:
        res += (v % 2) * i
        v //= 2; i *= 10
    return res

x = int(input())
v = binary((x//2)*2); cnt = 0
while v:
    if v % 10 == 1: cnt += 1
    v //= 10
print(cnt + (x % 2))
```
