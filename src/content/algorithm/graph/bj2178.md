[https://www.acmicpc.net/problem/2178](https://www.acmicpc.net/problem/2178)

(1, 1)에서부터 (n, m)까지 1을 거쳐서 몇번만에 갈 수 있는지 탐색하는 문제이다.
탐색과정에서 bfs를 사용하였다.

상하좌우 중 1값을 가진 칸을 찾고, 있다면 그 칸의 좌표를 큐에 넣고 (현재 칸의 깊이+1) 값을 찾은 칸의 좌표와 일치하는 depth배열 좌표에 저장한다.

최종적으로 depth배열의 (n, m)위치의 값을 출력.

(이때 maze배열의 크기가 가로세로 1씩 더 크게 만들었는데 그 이유는 탐색을 진행할 때 배열 범위 밖의 좌표를 탐색하는 경우를 없애기 위해서이다.)

이때 주의해야할 것은 입력값이 공백을 사이에 두고 주어지지 않는다는 것이다. 따라서 C언어같이 변수의 크기가 정해져있는 경우 숫자를 문자값으로 받는 것이 좋다.

C

```c
#include <stdio.h>

#define MaxSize 1000000
int maze[102][102];
int check[102][102];
int depth[102][102];
int d[2] = {-1, 1};

typedef struct{
    int data[MaxSize];
    int len;
    int front;
    int rear;
} Queue;

void initQueue(Queue* q){
    q->len = 0;
    q->front = 0;
    q->rear = 0;
}

int popleft(Queue* q){
    if(q->len > 0){
        int res = q->data[q->front];
        q->front = (q->front + 1) % MaxSize;
        q->len -= 1;
        return res;
    }
    return -1;
}

void append(Queue* q, int v){
    q->data[q->rear] = v;
    q->rear = (q->rear + 1) % MaxSize;
    q->len += 1;
}

void BFS(int x, int y){
    Queue q;
    initQueue(&q);
    check[y][x] = 1;
    append(&q, y);
    append(&q, x);

    while(q.len > 0){
        int cur_y = popleft(&q);
        int cur_x = popleft(&q);

        for(int i = 0; i < 2; i++){
            if(maze[cur_y + d[i]][cur_x] == 1 && check[cur_y + d[i]][cur_x] == 0){
                append(&q, cur_y + d[i]);
                append(&q, cur_x);
                depth[cur_y + d[i]][cur_x] = depth[cur_y][cur_x] + 1;
                check[cur_y + d[i]][cur_x] = 1;
            }
        }
        for(int i = 0; i < 2; i++){
            if(maze[cur_y][cur_x + d[i]] == 1 && check[cur_y][cur_x + d[i]] == 0){
                append(&q, cur_y);
                append(&q, cur_x + d[i]);
                depth[cur_y][cur_x + d[i]] = depth[cur_y][cur_x] + 1;
                check[cur_y][cur_x + d[i]] = 1;
            }
        }
    }
}

int main() {
    int n, m; scanf("%d %d", &n, &m);

    for(int i = 1; i <= n; i++){
        for(int j = 1; j <= m; j++){
            char c; scanf(" %c", &c);
            if(c == '1') maze[i][j] = 1;
        }
    }
    depth[1][1] = 1;
    BFS(1, 1);
    printf("%d", depth[n][m]);
    return 0;
}
```

Python

```python
from collections import deque
import sys

d = [-1, 1]

def bfs(x, y):
    q = deque()
    q.append([y, x])
    check[y][x] = True
    depth[y][x] = 1

    while q:
        v = q.popleft()

        for i in range(2):
            if maze[v[0] + d[i]][v[1]] == '1' and not check[v[0] + d[i]][v[1]]:
                q.append([v[0] + d[i], v[1]])
                depth[v[0] + d[i]][v[1]] = depth[v[0]][v[1]] + 1
                check[v[0] + d[i]][v[1]] = True
        for i in range(2):
            if maze[v[0]][v[1] + d[i]] == '1' and not check[v[0]][v[1] + d[i]]:
                q.append([v[0], v[1] + d[i]])
                depth[v[0]][v[1] + d[i]] = depth[v[0]][v[1]] + 1
                check[v[0]][v[1] + d[i]] = True

n, m = map(int, sys.stdin.readline().split())
maze = []
maze.append(['0' for _ in range(m + 2)])
for i in range(n):
    v = list(sys.stdin.readline().strip())
    v.insert(0, '0')
    v.append('0')
    maze.append(v)
maze.append(['0' for _ in range(m + 2)])

check = [[False for _ in range(m + 2)] for _ in range(n + 2)]
depth = [[0 for _ in range(m + 2)] for _ in range(n + 2)]
bfs(1, 1)

print(depth[n][m])
```
