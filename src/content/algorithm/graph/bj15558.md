왼쪽 1번째 칸부터 탐색 시작

현재 위치에서 할 수 있는 행동은 3가지
앞으로 한칸
뒤로 한칸(갈수록 칸이 없어지는 것을 유의)
대각선(다른 줄) 방향 k칸

이 세가지의 경우에 대해서 bfs로 계속 탐색해나감

위 세가지 경우를 보았을 때 각 행동에 따라 1에 도착하는지 0에 도착하는지 검사
1에 도착한다면(그리고 이 위치가 아직 탐색하지 않은 곳이라면) 큐에 해당 위치(줄과 칸 수) 그리고 깊이(depth)를 큐에 저장한다.

위와 같은 방식으로 탐색을 진행해나가다가 현재 칸에서 앞으로 한 칸 혹은 대각선(다른 줄) 방향으로 k칸 이동 시 n칸을 넘어서게 된다면 게임을 클리어 할 수 있는 것이므로 1을 출력.

탐색을 진행해도 n칸을 넘는 경우가 생길 수 없다면 0을 출력한다.

C

```c
#include <stdio.h>

#define MaxSize 1000000
char l[2][100001];
int check[2][100001];

typedef struct {
    int line;
    int idx;
    int depth;
} Node;

typedef struct {
    Node data[MaxSize];
    int front, rear, len;
} Queue;

void initQueue(Queue* q) {
    q->front = q->rear = q->len = 0;
}

void append(Queue* q, int line, int idx, int depth) {
    q->data[q->rear].line = line;
    q->data[q->rear].idx = idx;
    q->data[q->rear].depth = depth;
    q->rear = (q->rear + 1) % MaxSize;
    q->len++;
}

Node popleft(Queue* q) {
    Node res = q->data[q->front];
    q->front = (q->front + 1) % MaxSize;
    q->len--;
    return res;
}

int BFS(int n, int k) {
    Queue q;
    initQueue(&q);
    append(&q, 0, 0, 0);

    while (q.len > 0) {
        Node cur = popleft(&q);
        int line = cur.line;
        int idx = cur.idx;
        int depth = cur.depth;

        if (check[line][idx]) continue;
        check[line][idx] = 1;

        int next_idx[3] = {idx + 1, idx - 1, idx + k};
        int next_line[3] = {line, line, 1 - line};

        for (int i = 0; i < 3; i++) {
            int ni = next_idx[i];
            int nl = next_line[i];

            if (ni >= n) return 1;

            if (ni > depth && l[nl][ni] == '1' && !check[nl][ni]) {
                append(&q, nl, ni, depth + 1);
            }
        }
    }
    return 0;
}

int main() {
    int n, k;
    scanf("%d %d", &n, &k);
    for (int i = 0; i < 2; i++) {
        scanf("%s", l[i]);
    }
    printf("%d\n", BFS(n, k));
    return 0;
}
```

Python

```python
import sys
from collections import deque

def bfs():
    n, k = map(int, sys.stdin.readline().split())
    l = [list(sys.stdin.readline().strip()) for _ in range(2)]
    check = [[False for _ in range(n)] for _ in range(2)]
    
    q = deque([(0, 0, 0)]) # line, idx, depth
    
    while q:
        line, idx, depth = q.popleft()
        
        if check[line][idx]:
            continue
        check[line][idx] = True
        
        # 이동 가능한 3가지 경로
        for nl, ni in [(line, idx + 1), (line, idx - 1), (1 - line, idx + k)]:
            # n번 칸을 넘어선 경우
            if ni >= n:
                return 1
            
            # 이동 가능한 조건: 아직 사라지지 않은 칸(ni > depth)이고, 안전한 칸('1')일 때
            if ni > depth and l[nl][ni] == '1' and not check[nl][ni]:
                q.append((nl, ni, depth + 1))
                
    return 0

print(bfs())
```
