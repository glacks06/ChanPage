세 물통의 용량이 주어졌을 때, 첫 번째 물통(A)이 비어 있을 때 세 번째 물통(C)에 담길 수 있는 모든 물의 양을 구하는 문제이다.

각 물통에서 다른 물통으로 물을 붓는 6가지 경우의 수를 **DFS**로 탐색하며, 중복된 상태를 방지하기 위해 방문 여부를 체크하는 것이 핵심이다. 물을 부을 때 받는 물통이 넘치거나 주는 물통이 빌 때까지 붓는 조건을 수식으로 구현하였다.

-----

### C

```c
#include <stdio.h>
#include <stdlib.h>

int check[201][201];
int res[201];
int cnt = 0;

int compar(const void *a, const void *b) {
    return (*(int*)a - *(int*)b);
}

void append(int v) {
    for (int i = 0; i < cnt; i++) {
        if (res[i] == v) return;
    }
    res[cnt++] = v;
}

void dfs(int v1, int v2, int v3, int a, int b, int c) {
    if (check[v1][v2] == 1) return;
    if (v1 == 0) {
        append(v3);
    }
    check[v1][v2] = 1;

    // A -> B
    dfs((v1 + v2 < b) ? 0 : v1 - (b - v2), (v1 + v2 < b) ? v2 + v1 : b, v3, a, b, c);
    // A -> C
    dfs((v1 + v3 < c) ? 0 : v1 - (c - v3), v2, (v1 + v3 < c) ? v3 + v1 : c, a, b, c);
    // B -> A
    dfs((v2 + v1 < a) ? v1 + v2 : a, (v2 + v1 < a) ? 0 : v2 - (a - v1), v3, a, b, c);
    // B -> C
    dfs(v1, (v2 + v3 < c) ? 0 : v2 - (c - v3), (v2 + v3 < c) ? v2 + v3 : c, a, b, c);
    // C -> A
    dfs((v3 + v1 < a) ? v1 + v3 : a, v2, (v3 + v1 < a) ? 0 : v3 - (a - v1), a, b, c);
    // C -> B
    dfs(v1, (v3 + v2 < b) ? v2 + v3 : b, (v3 + v2 < b) ? 0 : v3 - (b - v2), a, b, c);
}

int main() {
    int a, b, c;
    scanf("%d %d %d", &a, &b, &c);
    dfs(0, 0, c, a, b, c);

    qsort(res, cnt, sizeof(int), compar);
    for (int i = 0; i < cnt; i++) printf("%d ", res[i]);
    
    return 0;
}
```

### Python

```python
import sys
sys.setrecursionlimit(10**6)

def dfs(v1, v2, v3):
    if check[v1][v2]:
        return
    if v1 == 0:
        res.add(v3)
    check[v1][v2] = True

    # A -> B
    if v1 > 0 and v2 < b:
        water = min(v1, b - v2)
        dfs(v1 - water, v2 + water, v3)
    # A -> C
    if v1 > 0 and v3 < c:
        water = min(v1, c - v3)
        dfs(v1 - water, v2, v3 + water)
    # B -> A
    if v2 > 0 and v1 < a:
        water = min(v2, a - v1)
        dfs(v1 + water, v2 - water, v3)
    # B -> C
    if v2 > 0 and v3 < c:
        water = min(v2, c - v3)
        dfs(v1, v2 - water, v3 + water)
    # C -> A
    if v3 > 0 and v1 < a:
        water = min(v3, a - v1)
        dfs(v1 + water, v2, v3 - water)
    # C -> B
    if v3 > 0 and v2 < b:
        water = min(v3, b - v2)
        dfs(v1, v2 + water, v3 - water)

a, b, c = map(int, sys.stdin.readline().split())
res = set()
# A와 B의 상태만 알면 C는 자동으로 결정되므로 2차원 배열로 체크 가능
check = [[False for _ in range(b + 1)] for _ in range(a + 1)]

dfs(0, 0, c)

for x in sorted(list(res)):
    print(x, end=' ')
```
