[https://www.acmicpc.net/problem/1389](https://www.acmicpc.net/problem/1389)

DFS 혹은 BFS를 이용하여 풀 수 있는 문제이며 나는 BFS를 선택하였다. (이유는 BFS방식을 좀 더 익숙하게 만들고 싶어서...)

입력으로 주어진 정점과 간선 정보를 이용해서 각 정점에서 탐색을 진행할 때 탐색 깊이를 배열에 따로 저장한 후 탐색을 진행한 BFS함수의 결과값으로 깊이를 저장한 배열의 모든 수를 합한 수를 내보냈다.

이 합한 수가 문제에서 말한 '케빈 베이컨의 수'가 된다.

결과적으로 함수가 반환한 모든 케빈 베이컨의 수 중 최소값을 출력하면 된다.

C

```c
#include <stdio.h>

#define MaxSize 101

int edge[101][5000];
int depth[MaxSize];
int check[MaxSize];

void init(){
    for(int i = 0; i < MaxSize; i++){
        depth[i] = 0;
        check[i] = 0;
    }
}

typedef struct{
    int data[MaxSize];
    int len;
    int front;
    int rear;
} Queue;

void initQueue(Queue* q){
    q->len = 0;
    q->front = 0;
    q->rear = 0;
}

int popleft(Queue* q){
    if(q->len > 0){
        int res = q->data[q->front];
        q->front += 1;
        q->len -= 1;
        if(q->front == MaxSize) q->front = 0;
        return res;
    }
}

void append(Queue* q, int v){
    int idx = q->rear;
    q->data[idx] = v;
    q->rear += 1;
    q->len += 1;
    if(q->rear == MaxSize) q->rear = 0;
}

int edgeAppend(int idx, int v){
    for(int i = 0; i < 101; i++){
        if(edge[idx][i] == 0) {edge[idx][i] = v; return 0;}
    }
}

int BFS(int v){
    Queue q;
    initQueue(&q);
    append(&q, v);
    check[v] = 1;

    while(q.len > 0){
        int a = popleft(&q);

        for(int i = 0; i < 5000; i++){
            if(edge[a][i] == 0) break;
            if(check[edge[a][i]] == 0){
                append(&q, edge[a][i]);
                check[edge[a][i]] = 1;
                depth[edge[a][i]] = depth[a]+1;
            }
        }
    }
    int res = 0;
    for(int i = 0; i < 101; i++)res += depth[i];
    return res;
}

int main() {
    int n, m; scanf("%d %d", &n, &m);

    for(int i = 0; i < m; i++){
        int a, b; scanf("%d %d", &a, &b);
        edgeAppend(a, b);
        edgeAppend(b, a);
    }
    int min = 100000;
    int idx = 0;
    for(int i = 1; i < n+1; i++){
        init();
        int t = BFS(i);
        if(t < min){min = t;idx = i;}
    }
    printf("%d", idx);
}
```

Python

```python
from collections import deque
import sys
sys.setrecursionlimit(10 ** 6)

def BFS(v):
    q = deque()
    q.append(v)
    check[v] = True
    while len(q) > 0:
        a = q.popleft()
        for i in edge[a]:
            if check[i] == False:
                q.append(i)
                check[i] = True
                depth[i] = depth[a]+1
    res = 0
    for i in depth: res += i
    return res

n, m = map(int, sys.stdin.readline().split())
edge = [[] for _ in range(n+1)]
for i in range(m):
    a, b = map(int, sys.stdin.readline().split())
    edge[a].append(b)
    edge[b].append(a)

min = 100000
idx = 0
check = []
depth = []

for i in range(1, n+1):
    check = [False for _ in range(n + 1)]
    depth = [0 for _ in range(n + 1)]
    t = BFS(i)
    if t < min: min = t; idx = i
print(idx)
```
