[https://www.acmicpc.net/problem/16948](https://www.acmicpc.net/problem/16948)

크기가 nxn인 배열(체스판))을 만든 후 모든 요소를 0으로 만든다

그 후 [r1, c1]부터 bfs로 탐색
데스나이트가 움직일 수 있는 범위를 참고하여 배열을 탐색하고 탐색한 값이 0이라면 아직 탐색하지 않은 곳임을 의미하므로
탐색한 좌표를 큐에 넣은 후 탐색 깊이를 배열(체스판)에 저장한다.

탐색 중 [r2, c2]에 도달하면 탐색 정지

배열에 저장한 [r2, c2]의 탐색 깊이를 출력하여 문제를 해결한다.

C

```c
#include <stdio.h>

#define MaxSize 100000

int l[200][200];

typedef struct{
    int data[MaxSize];
    int len;
    int front;
    int rear;
} Queue;

void initQueue(Queue* q){
    q->len = 0;
    q->front = 0;
    q->rear = 0;
}

int popleft(Queue* q){
    if(q->len > 0){
        int res = q->data[q->front];
        q->front += 1;
        q->len -= 1;
        if(q->front == MaxSize) q->front = 0;
        return res;
    }
    return -1;
}

void append(Queue* q, int v){
    int idx = q->rear;
    q->data[idx] = v;
    q->rear += 1;
    q->len += 1;
    if(q->rear == MaxSize) q->rear = 0;
}

void BFS(int n, int r1, int c1, int r2, int c2){
    Queue q;
    initQueue(&q);
    append(&q, r1);
    append(&q, c1);

    while(q.len > 0){
        int r = popleft(&q);
        int c = popleft(&q);
        if(r == r2 && c == c2) break;

        for(int i = -1; i <= 1; i += 2){
            if((r-2 >= 0) && (c+i >= 0 && c+i < n)){
                if(l[r-2][c+i] == 0){
                    append(&q, r-2); append(&q, c+i);
                    l[r-2][c+i] = l[r][c]+1;
                }
            }
        }
        for(int i = -2; i <= 2; i += 4){
            if(c+i >= 0 && c+i < n){
                if(l[r][c+i] == 0){
                    append(&q, r); append(&q, c+i);
                    l[r][c+i] = l[r][c]+1;
                }
            }
        }
        for(int i = -1; i <= 1; i += 2){
            if((r+2 < n) && (c+i >= 0 && c+i < n)){
                if(l[r+2][c+i] == 0) {
                    append(&q, r+2); append(&q, c+i);
                    l[r+2][c+i] = l[r][c]+1;
                }
            }
        }
    }
}

int main() {
    int n; scanf("%d", &n);
    int r1, c1, r2, c2; scanf("%d %d %d %d", &r1, &c1, &r2, &c2);
    BFS(n, r1, c1, r2, c2);
    if (r1 == r2 && c1 == c2) printf("0");
    else printf("%d", (l[r2][c2] > 0) ? l[r2][c2] : -1);
    return 0;
}
```

```python
import sys
from collections import deque

def bfs():
    q = deque()
    q.append([r1, c1])

    while q:
        r, c = q.popleft()
        if r == r2 and c == c2:
            return

        for i in [-1, 1]:
            if (r-2 >= 0) and (0 <= c+i < n):
                if l[r-2][c+i] == 0:
                    q.append([r-2, c+i])
                    l[r-2][c+i] = l[r][c]+1
        for i in [-2, 2]:
            if 0 <= c+i < n:
                if l[r][c+i] == 0:
                    q.append([r, c+i])
                    l[r][c+i] = l[r][c]+1
        for i in [-1, 1]:
            if (r+2 < n) and (0 <= c+i < n):
                if l[r+2][c+i] == 0:
                    q.append([r+2, c+i])
                    l[r+2][c+i] = l[r][c] + 1

n = int(sys.stdin.readline())
r1, c1, r2, c2 = map(int, sys.stdin.readline().split())

l = [[0 for _ in range(n)] for _ in range(n)]
bfs()

if r1 == r2 and c1 == c2:
    print(0)
else:
    print(l[r2][c2] if l[r2][c2] > 0 else -1)
```
