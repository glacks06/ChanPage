[https://www.acmicpc.net/problem/2331](https://www.acmicpc.net/problem/2331)

$D[1] = A$, $D[n] = D[n-1]$의 각 자리의 숫자를 $P$번 곱한 수들의 합으로 정의된 수열에서 반복되는 부분을 제외하고 남은 수들의 개수를 구하는 문제이다.

수열을 따라가다 보면 이전에 나왔던 숫자가 다시 등장하게 되는데, 그 숫자가 나타나기 전까지의 수열 길이가 정답이 된다. 중복을 확인하기 위해 방문 여부를 저장하는 배열과 수열의 순서를 기록하는 리스트를 활용하여 해결할 수 있다.

C

```c
#include <stdio.h>

#define MaxSize 10000
int record[MaxSize];
int check[236197];

void append(int v){
    for(int i = 0; i < MaxSize; i++){
        if(record[i] == 0) {
            record[i] = v;
            break;
        }
    }
}

int DFS(int v, int p){
    check[v] = 1;
    append(v);
    int sum = 0;
    int temp = v;
    while(temp > 0){
        int mtp = 1;
        for(int i = 0; i < p; i++) mtp *= temp % 10;
        sum += mtp;
        temp /= 10;
    }
    if(check[sum] == 1) return sum;
    else return DFS(sum, p);
}

int main() {
    int a, p; scanf("%d %d", &a, &p);
    int v = DFS(a, p);
    int cnt = 0;
    for(int i = 0; i < MaxSize; i++){
        if(record[i] == v) {
            printf("%d", cnt);
            break;
        }
        cnt += 1;
    }
    return 0;
}
```

Python

```python
import sys
sys.setrecursionlimit(10 ** 6)

def DFS(v, p):
    check[v] = True
    record.append(v)
    
    next_v = 0
    for s in str(v):
        next_v += int(s) ** p
        
    if check[next_v]:
        return next_v
    else:
        return DFS(next_v, p)
    
record = []
a, p = map(int, sys.stdin.readline().split())
check = [False for _ in range(236197)]

v = DFS(a, p)
cnt = 0
for i in record:
    if i == v:
        print(cnt)
        break
    cnt += 1
```
