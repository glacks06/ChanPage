1번 칸부터 탐색 시작
1번칸에서 1\~6칸을 갔을 때 도착하는 칸을 큐에 넣기

1\~6칸을 갔을 때 사다리 혹은 뱁 이벤트를 반영해서 최종적인 도착지점을 큐에 넣어야 함(결국 뱀이냐 사다리냐가 아닌 어떤 칸에 갔을 때 어디로 가느냐가 중요하기 때문에 굳이 구분해서 저장할 필요 없음. 따라서 코드 속 ladderAndSnake처럼 한 (2차원)배열에 다 저장해도 됨)

큐에 넣으면서 현재 지점까지 오는데까지 굴린 주사위 횟수(깊이)를 배열에 기록

탐색을 마친 후 기록한 배열을 이용해서 결과값 출력

C

```c
#include <stdio.h>

#define MaxSize 10000

int move[101];
int check[101];
int dist[101];

typedef struct {
    int data[MaxSize];
    int front, rear, len;
} Queue;

void initQueue(Queue* q) {
    q->front = q->rear = q->len = 0;
}

void append(Queue* q, int v) {
    q->data[q->rear] = v;
    q->rear = (q->rear + 1) % MaxSize;
    q->len++;
}

int popleft(Queue* q) {
    int res = q->data[q->front];
    q->front = (q->front + 1) % MaxSize;
    q->len--;
    return res;
}

int main() {
    int n, m;
    scanf("%d %d", &n, &m);

    for (int i = 1; i <= 100; i++) move[i] = i;

    for (int i = 0; i < n + m; i++) {
        int u, v;
        scanf("%d %d", &u, &v);
        move[u] = v;
    }

    Queue q;
    initQueue(&q);
    append(&q, 1);
    check[1] = 1;
    dist[1] = 0;

    while (q.len > 0) {
        int cur = popleft(&q);

        if (cur == 100) {
            printf("%d\n", dist[100]);
            break;
        }

        for (int i = 1; i <= 6; i++) {
            int next = cur + i;
            if (next <= 100) {
                next = move[next];
                if (!check[next]) {
                    check[next] = 1;
                    dist[next] = dist[cur] + 1;
                    append(&q, next);
                }
            }
        }
    }

    return 0;
}
```

Python

```python
import sys
from collections import deque

def solve():
    n, m = map(int, sys.stdin.readline().split())
    
    # 사다리와 뱀 정보를 하나의 딕셔너리나 배열에 저장
    move = [i for i in range(101)]
    for _ in range(n + m):
        u, v = map(int, sys.stdin.readline().split())
        move[u] = v
        
    # BFS를 위한 큐와 방문 확인 배열
    q = deque([1])
    dist = [-1] * 101
    dist[1] = 0
    
    while q:
        cur = q.popleft()
        
        if cur == 100:
            print(dist[100])
            return

        # 주사위 1~6 눈금 확인
        for i in range(1, 7):
            next_pos = cur + i
            if next_pos <= 100:
                # 사다리나 뱀이 있다면 이동
                next_pos = move[next_pos]
                if dist[next_pos] == -1:
                    dist[next_pos] = dist[cur] + 1
                    q.append(next_pos)

solve()
```
