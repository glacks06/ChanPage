[https://www.acmicpc.net/problem/2667](https://www.acmicpc.net/problem/2667)

탐색을 마친 횟수가 곧 단지 수이다.

우선 입력받은 값을 반복문으로 하나씩 확인하면서 1을 찾는다.

1을 찾았다면 해당 위치에서부터 dfs시작한다.
dfs가 끝나면 단지 하나가 탐색 끝난것을 의미하므로 결과값에 +1
(탐색한 아파트 수 또한 저장해둔다.)

그 다음 아직 탐색을 진행하지 않은 1이 있는지 찾기

모든 1을 탐색할때가지 반복 후 결과 출력.

  + 
값을 입력받을 때 주의해야할 것은 입력값이 공백을 사이에 두고 주어지지 않는다는 것이다. 따라서 C언어같이 변수의 크기가 정해져있는 경우 숫자를 문자값으로 받는 것이 좋다.

C

```c
#include <stdio.h>
#include <stdlib.h>

int apt[30][30];
int check[30][30];
int res[625];
int aptCnt = 0;
int d[2] = {-1, 1};

int compar(const void *a, const void *b) {
    return (*(int*)a - *(int*)b);
}

void dfs(int x, int y, int n){
    if (check[y][x] == 1 || apt[y][x] == 0) return;

    check[y][x] = 1;
    aptCnt++;

    for(int i = 0; i < 2; i++){
        if(y+d[i] >= 1 && y+d[i] <= n) dfs(x, y+d[i], n);
        if(x+d[i] >= 1 && x+d[i] <= n) dfs(x+d[i], y, n);
    }
}

int main(){
    int n; scanf("%d", &n);

    for(int i = 1; i <= n; i++){
        for(int j = 1; j <= n; j++){
            char c; scanf(" %c", &c);
            if(c == '1') apt[i][j] = 1;
        }
    }

    int cnt = 0;
    for(int i = 1; i <= n; i++){
        for(int j = 1; j <= n; j++){
            if(apt[i][j] == 1 && check[i][j] == 0){
                aptCnt = 0;
                dfs(j, i, n);
                res[cnt++] = aptCnt;
            }
        }
    }
    
    qsort(res, cnt, sizeof(int), compar);

    printf("%d\n", cnt);
    for(int i = 0; i < cnt; i++) printf("%d\n", res[i]);
    return 0;
}
```

Python

```python
import sys
sys.setrecursionlimit(10**6)

n = int(sys.stdin.readline())
v = [list(sys.stdin.readline().strip()) for _ in range(n)]
check = [[False for _ in range(n)] for _ in range(n)]

dx = [0, 0, 1, -1]
dy = [1, -1, 0, 0]

def dfs(x, y):
    check[y][x] = True
    global aptCnt
    aptCnt += 1

    for i in range(4):
        nx, ny = x + dx[i], y + dy[i]
        if 0 <= nx < n and 0 <= ny < n:
            if v[ny][nx] == '1' and not check[ny][nx]:
                dfs(nx, ny)

cnt = 0
res = []

for i in range(n):
    for j in range(n):
        if v[i][j] == '1' and not check[i][j]:
            aptCnt = 0
            dfs(j, i)
            res.append(aptCnt)
            cnt += 1

res.sort()
print(cnt)
for out in res:
    print(out)
```
